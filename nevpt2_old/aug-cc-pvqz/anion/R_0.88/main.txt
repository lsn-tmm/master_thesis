outfile = open('results.txt','w')
mol = gto.M(atom=[['O',(0,0,0)],['H',(0,0,0.88)]],basis='aug-cc-pvqz',spin=0,charge=-1,

mf  = scf.ROHF(mol)

BO_IAO = BO_class(mol,mf)
